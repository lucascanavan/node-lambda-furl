# Sample AWS Lambda using Function URL, nodejs, typescript, terraform & lambda-api.

# Development
npm run dev

# Deploy ("terraform init" from "./terraform" will need to be run once before the first deploy)
npm run deploy

# Destroy (clean up)
npm run destroy
